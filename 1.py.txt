import pandas as p
import numpy as n
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score, classification_report
a = p.read_csv("dataset.csv")
print("Columns in the dataset:", a.columns)
a = a.drop(columns=["loan_id"])
print(a.head())
a.fillna(a.mean(),inplace=True)
x = a.drop(columns=["loan_status"], axis=1)
y = a["loan_status"]
x=p.get_dummies(x) 
x_train, x_test, y_train, y_test = train_test_split(x, y, test_size=0.2, random_state=42)
L=LogisticRegression()
L.fit(x_train,y_train)
y_predict=L.predict(x_test)
ac=accuracy_score(y_predict,y_test)
c=classification_report(y_predict,y_test)
print(ac)
print(c)
